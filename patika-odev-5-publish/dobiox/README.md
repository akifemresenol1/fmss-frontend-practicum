## Install

```bash
npm install --save dobiox
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'dobiox'
import 'dobiox/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
